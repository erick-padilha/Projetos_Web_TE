Exemplos de Escopos

Escopos determinam a visibilidade das variáveis e funções em diferentes partes do código.

            
                // Escopo global
                let global = "Estou no escopo global";

                function meuEscopo() {
                    // Escopo local
                    let local = "Estou no escopo local";
                    console.log(global); // Pode acessar variável global
                    console.log(local);  // Pode acessar variável local
                }

                meuEscopo();
                console.log(global); // Pode acessar variável global
                // console.log(local); // Erro! Não pode acessar variável local fora da função
